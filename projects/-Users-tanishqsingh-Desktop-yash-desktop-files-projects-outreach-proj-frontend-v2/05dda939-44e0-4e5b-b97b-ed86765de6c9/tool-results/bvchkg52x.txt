      <div style={{ background: "rgba(232,184,122,0.08)", border: "0.5px solid rgba(232,184,122,0.3)", borderRadius: 8, padding: 14, marginBottom: 18, display: "flex", gap: 10 }}>
        <Icon name="warn" size={16} style={{ color: "var(--amber)", flexShrink: 0, marginTop: 1 }} />
        <div style={{ flex: 1 }}>
          <div style={{ fontSize: 12.5, fontWeight: 550, color: "var(--amber)" }}>Heads up — emails sourced via scraping may be outdated</div>
          <div style={{ fontSize: 12, color: "var(--fg-1)", marginTop: 6, lineHeight: 1.55 }}>
            Scraped emails can be incorrect, role-based, or no longer monitored. Expect a bounce rate around 18–25% — significantly higher than Apollo. Repeated bounces can also affect your sender reputation.
          </div>
          <label style={{ display: "flex", alignItems: "center", gap: 8, marginTop: 12, cursor: "pointer", fontSize: 12 }}>
            <input type="checkbox" checked={data.scrapingAck} onChange={(e) => set("scrapingAck", e.target.checked)} style={{ accentColor: "var(--amber)" }} />
            <span>I understand the risks and want to proceed with web scraping.</span>
          </label>
        </div>
      </div>
    )}

    <div style={{ fontSize: 11, fontWeight: 500, letterSpacing: "0.06em", textTransform: "uppercase", color: "var(--fg-3)", marginBottom: 10, marginTop: 8 }}>AI provider</div>
    <Field label="Model" hint="Used for personalization, subject lines, and reply drafting.">
      <select className="field" value={data.llm} onChange={(e) => set("llm", e.target.value)}>
        <option value="gpt-5">OpenAI · GPT-5 (recommended)</option>
        <option value="claude">Anthropic · Claude Sonnet 4.5</option>
        <option value="local">Local · Llama 3.3 70B</option>
      </select>
    </Field>
    <Field label="Tone" hint="How should drafts sound?">
      <div style={{ display: "flex", gap: 6 }}>
        {["Direct", "Warm", "Formal", "Casual"].map((t, i) => (
          <button key={t} className="btn btn-sm" style={{ background: i === 0 ? "var(--accent-dim)" : "var(--bg-1)", borderColor: i === 0 ? "var(--accent)" : "var(--border-strong)", color: i === 0 ? "var(--accent)" : "var(--fg-1)", flex: 1, justifyContent: "center" }}>{t}</button>
        ))}
      </div>
    </Field>
  </>
);

const StepReview = ({ data, isB2B, setStep }: { data: WizardData; isB2B: boolean; setStep: (n: number) => void }) => {
  const rows = [
    { label: "Campaign type", value: isB2B ? "B2B sales outreach" : "Job outreach", step: 0 },
    { label: "Intent", value: data.intent, step: 0, multi: true },
    { label: "Companies", value: data.companies.split(",").length + " companies", step: 1 },
    { label: "Titles", value: (data.titles as string[]).join(", "), step: 1 },
    { label: "Regions", value: (data.regions as string[]).join(", "), step: 1 },
    { label: "Sender", value: data.sender, step: 2 },
    { label: "Email source", value: data.emailSource === "apollo" ? "Apollo API" : "Web scraping", step: 3, warn: data.emailSource === "scrape" },
    { label: "AI model", value: ({ "gpt-5": "OpenAI GPT-5", claude: "Claude Sonnet 4.5", local: "Llama 3.3 70B" } as Record<string, string>)[data.llm], step: 3 },
  ];
  return (
    <>
      <div style={{ background: "var(--bg-1)", border: "0.5px solid var(--border)", borderRadius: 10, overflow: "hidden", marginBottom: 18 }}>
        {rows.map((r, i) => (
          <div key={i} style={{ display: "flex", padding: "10px 14px", borderBottom: i < rows.length - 1 ? "0.5px solid var(--border)" : "none", gap: 12, alignItems: r.multi ? "flex-start" : "center" }}>
            <div style={{ width: 130, fontSize: 11.5, color: "var(--fg-2)", flexShrink: 0 }}>{r.label}</div>
            <div style={{ flex: 1, fontSize: 12.5, color: r.warn ? "var(--amber)" : "var(--fg)", lineHeight: 1.5, display: "flex", alignItems: "center", gap: 6 }}>
              {r.warn && <Icon name="warn" size={12} />}{r.value}
            </div>
            <button className="btn btn-ghost btn-sm" onClick={() => setStep(r.step)}>
              <Icon name="edit" size={11} />Edit
            </button>
          </div>
        ))}
      </div>
      <div style={{ background: "var(--bg-1)", border: "0.5px solid var(--accent)", borderRadius: 10, padding: 14, display: "flex", gap: 10, alignItems: "flex-start" }}>
        <Icon name="sparkles" size={15} style={{ color: "var(--accent)", marginTop: 2 }} />
        <div style={{ flex: 1 }}>
          <div style={{ fontSize: 12.5, fontWeight: 550 }}>Ready to launch</div>
          <div style={{ fontSize: 12, color: "var(--fg-2)", marginTop: 4, lineHeight: 1.55 }}>
            We'll find ~150 contacts, draft personalized emails for each, and assemble your sendable list. Estimated time: <span style={{ color: "var(--fg)", fontFamily: "var(--font-mono)" }}>3–5 min</span>.
          </div>
        </div>
      </div>
    </>
  );
};

export default function WizardPage() {
  const { id } = useParams<{ id: string }>();
  const router = useRouter();
  const [campaignType] = React.useState("b2b");
  const [step, setStep] = React.useState(0);
  const [data, setData] = React.useState<WizardData>({
    intent: "Pitch our observability platform to engineering leaders at Series B+ fintechs. Focus on teams suffering from MTTR pain.",
    body: "Hi {{firstName}},\n\nNoticed {{company}} just shipped real-time payments — congrats. Teams at that scale usually start hitting MTTR walls around 90 days in. We help engineering orgs cut MTTR by ~40% without ripping out their existing stack.\n\nWorth a 15-min look next week?\n\n— Jordan",
    companies: "Mercury, Ramp, Brex, Modern Treasury, Stripe Issuing, Plaid, Unit, Increase, Highnote, Lithic",
    regions: ["North America", "Europe"],
    titles: ["VP Engineering", "Head of Platform", "Director of Infrastructure"],
    companySize: "100-1000",
    industry: "Fintech / Payments",
    valueProp: "Cut MTTR by 40% with zero-instrumentation observability.",
    sender: "jordan@company.com",
    linkedin: "https://linkedin.com/in/jordan-marsh",
    emailSource: "apollo",
    scrapingAck: false,
    llm: "gpt-5",
  });

  const set = (k: keyof WizardData, v: unknown) => setData((d) => ({ ...d, [k]: v }));

  const isB2B = campaignType !== "job";
  const steps = isB2B
    ? [
        { id: "intent", title: "Intent", subtitle: "Why are you reaching out?" },
        { id: "target", title: "Target", subtitle: "Who should we contact?" },
        { id: "profile", title: "Sender", subtitle: "Who's sending these?" },
        { id: "settings", title: "Sources", subtitle: "Email + AI providers" },
        { id: "review", title: "Review", subtitle: "Confirm and launch" },
      ]
    : [
        { id: "intent", title: "Role", subtitle: "What are you looking for?" },
        { id: "target", title: "Targets", subtitle: "Where do you want to work?" },
        { id: "profile", title: "Profile", subtitle: "Resume + links" },
        { id: "settings", title: "Sources", subtitle: "Email + AI providers" },
        { id: "review", title: "Review", subtitle: "Confirm and launch" },
      ];

  const total = steps.length;
  const next = () => setStep((s) => Math.min(total - 1, s + 1));
  const prev = () => (step === 0 ? router.back() : setStep((s) => s - 1));

  const canAdvance = () => {
    if (step === 3 && data.emailSource === "scrape" && !data.scrapingAck) return false;
    return true;
  };

  const handleLaunch = async () => {
    await fetch(`/api/v1/campaigns/${id}/sessions`, {
      method: "PUT",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ extracted_params: data }),
    });
    router.push(`/campaign/${id}/processing`);
  };

  return (
    <div className="shell">
      <Sidebar active="new" />
      <div className="main">
        <Topbar
          crumb={
            <>
              <span className="muted">New campaign</span>
              <Icon name="chev_r" size={11} className="sep" style={{ color: "var(--fg-3)" }} />
              <span>{isB2B ? "B2B sales outreach" : "Job outreach"}</span>
              <span className="badge b-draft" style={{ marginLeft: 8 }}>
                <span className="badge-dot" />Draft
              </span>
            </>
          }
          right={
            <>
              <span style={{ fontSize: 11, color: "var(--fg-3)" }}>Auto-saved 2s ago</span>
              <button className="btn btn-ghost btn-sm">Save and exit</button>
            </>
          }
        />

        <div style={{ display: "flex", flex: 1, minHeight: 0 }}>
          {/* Step rail */}
          <div style={{ width: 220, borderRight: "0.5px solid var(--border)", padding: 18, flexShrink: 0 }}>
            <div style={{ fontSize: 10.5, fontWeight: 500, letterSpacing: "0.06em", textTransform: "uppercase", color: "var(--fg-3)", marginBottom: 14 }}>Setup · {step + 1}/{total}</div>
            <div style={{ display: "flex", flexDirection: "column", gap: 2 }}>
              {steps.map((s, i) => {
                const done = i < step;
                const active = i === step;
                return (
                  <div key={s.id} onClick={() => i <= step && setStep(i)} style={{ display: "flex", gap: 10, padding: "8px 8px", borderRadius: 6, background: active ? "var(--bg-2)" : "transparent", cursor: i <= step ? "pointer" : "default", alignItems: "flex-start" }}>
                    <div style={{ width: 18, height: 18, borderRadius: "50%", background: done ? "var(--accent)" : active ? "var(--bg-3)" : "var(--bg-2)", border: active ? "1px solid var(--accent)" : "0.5px solid var(--border-strong)", display: "grid", placeItems: "center", flexShrink: 0, marginTop: 1 }}>
                      {done ? <Icon name="check" size={11} stroke={2.4} style={{ color: "var(--accent-fg)" }} /> : <span style={{ fontSize: 10, fontWeight: 600, color: active ? "var(--accent)" : "var(--fg-3)", fontFamily: "var(--font-mono)" }}>{i + 1}</span>}
                    </div>
                    <div style={{ minWidth: 0 }}>
                      <div style={{ fontSize: 12.5, fontWeight: 500, color: active ? "var(--fg)" : done ? "var(--fg-1)" : "var(--fg-2)" }}>{s.title}</div>
                      <div style={{ fontSize: 11, color: "var(--fg-3)", marginTop: 1 }}>{s.subtitle}</div>
                    </div>
                  </div>
                );
              })}
            </div>

            <div style={{ marginTop: 20, padding: 12, background: "var(--bg-1)", border: "0.5px solid var(--border)", borderRadius: 8 }}>
              <div style={{ display: "flex", alignItems: "center", gap: 6, fontSize: 11, color: "var(--fg-2)" }}>
                <Icon name="info" size={12} />Tip
              </div>
              <div style={{ fontSize: 11.5, color: "var(--fg-2)", marginTop: 6, lineHeight: 1.5 }}>
                You can paste raw notes anywhere. We'll extract structured fields from them automatically.
              </div>
            </div>
          </div>

          {/* Step content */}
          <div style={{ flex: 1, overflow: "auto", padding: "32px 48px 120px", position: "relative" }}>
            <div style={{ maxWidth: 640, margin: "0 auto" }}>
              <div style={{ fontSize: 10.5, fontWeight: 500, letterSpacing: "0.06em", textTransform: "uppercase", color: "var(--fg-3)" }}>Step {step + 1} of {total}</div>
              <h2 style={{ fontSize: 22, fontWeight: 600, letterSpacing: "-0.01em", margin: "6px 0 4px" }}>{steps[step].title}</h2>
              <p style={{ fontSize: 13.5, color: "var(--fg-2)", margin: "0 0 24px" }}>{steps[step].subtitle}</p>

              {step === 0 && <StepIntent data={data} set={set} isB2B={isB2B} />}
              {step === 1 && <StepTarget data={data} set={set} isB2B={isB2B} />}
              {step === 2 && <StepProfile data={data} set={set} isB2B={isB2B} />}
              {step === 3 && <StepSettings data={data} set={set} />}
              {step === 4 && <StepReview data={data} isB2B={isB2B} setStep={setStep} />}
            </div>
          </div>
        </div>

        {/* Footer nav */}
        <div style={{ height: 56, borderTop: "0.5px solid var(--border)", display: "flex", alignItems: "center", padding: "0 24px", flexShrink: 0, gap: 10, background: "var(--bg)" }}>
          <button className="btn btn-sm" onClick={prev}>
            <Icon name="arrow_l" size={12} />
            {step === 0 ? "Cancel" : "Back"}
          </button>
          <div style={{ flex: 1, display: "flex", alignItems: "center", justifyContent: "center", gap: 4 }}>
            {steps.map((_, i) => (
              <div key={i} style={{ width: 22, height: 3, borderRadius: 2, background: i <= step ? "var(--accent)" : "var(--bg-3)" }} />
            ))}
          </div>
          {step < total - 1 ? (
            <button className="btn btn-primary btn-sm" onClick={next} disabled={!canAdvance()} style={{ opacity: canAdvance() ? 1 : 0.5 }}>
              Continue
              <Icon name="arrow_r" size={12} />
            </button>
          ) : (
            <button className="btn btn-primary btn-sm" onClick={handleLaunch}>
              <Icon name="sparkles" size={12} />
              Launch campaign
            </button>
          )}
        </div>
      </div>
    </div>
  );
}
